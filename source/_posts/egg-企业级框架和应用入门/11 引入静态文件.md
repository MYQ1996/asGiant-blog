exports.ejs = {

### 11.2 配置设置
> a：静态文件


```javascript
config.static = {

      prefix: '/',

      dir: path.join(appInfo.baseDir, 'app/public/')

    }
```

> 当然此时你会遇到一个问题，你想要多个文件该如何事好


```javascript
 config.static = {
    prefix: '/',
    dir: [ path.join(appInfo.baseDir, 'app/view/'),
      path.join(appInfo.baseDir, 'app/public/uploads/'),
      path.join(appInfo.baseDir, 'app/public/swagger/') ],
  };
```

 b：模板设置

```javascript
config.view = {
  defaultExt: '.html',
  mapping: {
    '.ejs': 'ejs',
    '.html': 'ejs',
  }
}
```
